Programa que lê a idade de três pessoas e indica quem é mais velha, do meio e mais nova.


#include <stdio.h>

int main(void) {
    int ci1, ca2, c3, inteiro;

    printf("\nColoque a idade de Cibele:");
    scanf("%d", &ci1);

    printf("\nColoque a idade de Camila:");
    scanf("%d", &ca2);

    printf("\nColoque a idade de Celeste:");
    scanf("%d", &c3);

    if ((ci1 > ca2) && (ci1 > c3)) {
        printf("\nCibele e mais velha");
    }

    if ((ca2 < ci1) && (ca2 > c3)) {
        printf("\nCamila e a do meio");
    }

    if ((c3 < ci1) && (c3 < ca2)) {
        printf("\nCeleste e a mais nova");
    }

    if ((ci1 == ca2) && (ci1 == c3) && (c3 == ca2)) {
        printf("\nTodas tem a mesma idade");
    }

    if ((ci1 > ca2) && (c3 > ca2)) {
        printf("\nCeleste e a do meio");
    }
}
